#MODIFIED SYNTAX REPLACING 'ALTER' (WIP)

    '-' FOR TABLE
    'VALUES' TO GO INSIDE THE LEVEL
    NO 'VALUES' TO UPDATE THE LEVEL ON SURFACE

    Y UPDATE database_name new_dbname     #Changes the database_name
    Y UPDATE database_name VALUES newtable_2       #Changes the table name (_ for no changes)
    Y UPDATE -tablename newid, newname     #Changes the columnname (cannot change existing schema datatype)
    UPDATE -tablename VALUES (1, Shauyra),(2, Varun),....
    OR
    UPDATE -tablename VALUES (1, Shauyra) WHERE id==all   #Changes the column values (can only modify one when used WHERE)

    errors:
    THERE ARE 2 TABLES BUT 3 ARE MENTIONED IN CHANGE TABLE COMMAND (IF THERE WERE 1 IT SHOULD UPDATE FIRST TABLE ONLY)
    SAME THINGS WITH COLUMN NAMES
    SAME THINGS WITH COLUMN VALUES

#CREATE SYNTAX  

    '_' CANNOT BE USED FOR CREATE AS YOU CAN'T SKIP WHAT DOESN'T EXIST YET

    CREATE database_name (DB WILL NOT BE INCLUDED ANYMORE AS HYPHEN WILL DICTATE IF IT'S DATABASE OR TABLE.)
    CREATE database_name VALUES tabl1,tabl2,... (CREATES TABLE AS VALUES ALLOW TO DIVE DEEPER IN DATABASE, HENCE CREATING TABLE)
    CREATE -tb_name val1:dtype, val2:dtype.... (THIS WILL BE USED TO CREATE NEW COLUMN IN EXISITING TABLES)
    CREATE -tb_name VALUES (va1,val2),(val1,val2).... (USING VALUES WILL GO INSIDE TABLE AND CREATE ROWS)
